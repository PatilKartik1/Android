package com.example.fontchanger;

import android.graphics.Color;
import android.graphics.Typeface;
import android.os.Bundle;
import android.widget.Button;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    private TextView tvSampleText;
    private float currentFontSize = 20f;
    private boolean isBold = false;
    private boolean isColorChanged = false;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        tvSampleText = findViewById(R.id.tvSampleText);
        Button btnIncreaseSize = findViewById(R.id.btnIncreaseSize);
        Button btnChangeColor = findViewById(R.id.btnChangeColor);
        Button btnChangeFont = findViewById(R.id.btnChangeFont);

        btnIncreaseSize.setOnClickListener(v -> {
            currentFontSize += 5;
            tvSampleText.setTextSize(currentFontSize);
        });

        btnChangeColor.setOnClickListener(v -> {
            if (isColorChanged) {
                tvSampleText.setTextColor(Color.BLACK);
            } else {
                tvSampleText.setTextColor(Color.RED);
            }
            isColorChanged = !isColorChanged;
        });

        btnChangeFont.setOnClickListener(v -> {
            if (isBold) {
                tvSampleText.setTypeface(Typeface.SANS_SERIF, Typeface.NORMAL);
            } else {
                tvSampleText.setTypeface(Typeface.SERIF, Typeface.BOLD);
            }
            isBold = !isBold;
        });
    }
}
